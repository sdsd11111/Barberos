<script>
  // Memoria interna por si el navegador bloquea el localStorage
  let memoryStorage = [];
  let selectedAvatar = '';

  // Cargar historial al inicio
  try {
    const saved = localStorage.getItem('barberos_sessions');
    if (saved) memoryStorage = JSON.parse(saved);
  } catch(e) { console.log("LocalStorage bloqueado, usando memoria temporal"); }

  updateHistCount();

  function toggleField(id, btnId) {
    const el = document.getElementById(id);
    const btn = document.getElementById(btnId);
    el.classList.toggle('hidden');
    if (!el.classList.contains('hidden')) btn.classList.add('active');
    else btn.classList.remove('active');
  }

  function selectAvatar(btn) {
    document.querySelectorAll('.avatar-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    selectedAvatar = btn.dataset.val;
  }

  function saveSession() {
    const data = {
      barberName: document.getElementById('barberName').value,
      p0: document.getElementById('p0').value,
      p1: document.getElementById('p1').value,
      p2: document.getElementById('p2').value,
      p2_resp: document.getElementById('p2_resp').value,
      p3: document.getElementById('p3').value,
      p4a: document.getElementById('p4a').value,
      p4b: document.getElementById('p4b').value,
      p5: document.getElementById('p5').value,
      p6: document.getElementById('p6').value,
      avatar: selectedAvatar,
      notes: document.getElementById('notes').value
    };

    if (!data.barberName && !data.p0) {
      alert("Escribe al menos el nombre o la primera respuesta.");
      return;
    }

    data.id = Date.now();
    data.date = new Date().toLocaleString('es-CO');
    
    // Guardar en memoria
    memoryStorage.unshift(data);
    
    // Intentar guardar permanente
    try {
      localStorage.setItem('barberos_sessions', JSON.stringify(memoryStorage));
    } catch(e) {
      alert("Tu navegador está bloqueando el guardado permanente. La sesión se guardó solo por esta sesión.");
    }
    
    updateHistCount();

    // Feedback visual
    const btn = document.getElementById('saveBtn');
    btn.innerText = "¡GUARDADO!";
    btn.style.background = "#10b981";
    btn.style.color = "white";
    if(navigator.vibrate) navigator.vibrate(200);
    
    setTimeout(() => {
      btn.innerText = "GUARDAR SESIÓN";
      btn.style.background = "#FF6B1A";
      btn.style.color = "black";
      clearForm();
    }, 1500);
  }

  function clearForm() {
    document.querySelectorAll('textarea, input').forEach(el => el.value = '');
    document.querySelectorAll('.toggle-btn').forEach(b => b.classList.remove('active'));
    document.querySelectorAll('.hidden').forEach(el => el.classList.add('hidden')); 
    selectedAvatar = '';
  }

  function updateHistCount() {
    document.getElementById('histCount').innerText = memoryStorage.length;
  }

  function openHistory() {
    const list = document.getElementById('histList');
    
    if (memoryStorage.length === 0) {
      list.innerHTML = '<p class="text-gray-500 text-center mt-10">No hay sesiones guardadas.</p>';
    } else {
      list.innerHTML = memoryStorage.map(s => `
        <div class="hist-item" onclick="viewDetail(${s.id})">
          <div class="flex justify-between items-center">
            <div class="font-bold text-base">${s.barberName || 'Sin nombre'}</div>
            <div class="text-xs text-gray-500">${s.date}</div>
          </div>
          <div class="text-sm text-gray-400 mt-1 truncate">
            ${s.avatar ? `<span class="text-orange-500 font-bold uppercase">${s.avatar}</span> · ` : ''}
            ${s.p0 ? s.p0.substring(0, 40) + '...' : 'Sin respuesta'}
          </div>
        </div>
      `).join('');
    }
    document.getElementById('histModal').classList.remove('hidden');
  }

  function closeHistory() {
    document.getElementById('histModal').classList.add('hidden');
  }

  function viewDetail(id) {
    const s = memoryStorage.find(x => x.id == id);
    if (!s) return;
    
    let html = <div class="mb-6"><strong class="text-orange-500 text-xl block">${s.barberName}</strong><small class="text-gray-500">${s.date}</small></div>;
    
    const fields = [
      ['P0', s.p0], ['P1', s.p1], ['P2', s.p2], ['P2 Cruce', s.p2_resp],
      ['P3', s.p3], ['P4A', s.p4a], ['P4B', s.p4b], ['P5', s.p5],
      ['P6', s.p6], ['Avatar', s.avatar], ['Notas', s.notes]
    ];
    
    fields.forEach(([label, val]) => {
      if (val) {
        html += <div class="mb-4"><div class="text-xs text-gray-500 uppercase font-bold mb-1">${label}</div><div class="bg-[#141414] p-3 rounded text-sm">${val}</div></div>;
      }
    });
    
    html += `<div class="flex gap-2 mt-8">
      <button onclick="deleteSession(${id})" class="bg-red-900 text-red-200 px-4 py-3 rounded-lg text-sm font-bold">Eliminar</button>
      <button onclick="openHistory()" class="bg-gray-800 text-gray-300 px-4 py-3 rounded-lg text-sm font-bold">Volver</button>
    </div>`;
    
    document.getElementById('histList').innerHTML = html;
  }

  function deleteSession(id) {
    if (!confirm("¿Eliminar esta sesión?")) return;
    memoryStorage = memoryStorage.filter(s => s.id != id);
    try {
      localStorage.setItem('barberos_sessions', JSON.stringify(memoryStorage));
    } catch(e) {}
    updateHistCount();
    openHistory();
  }
</script>
